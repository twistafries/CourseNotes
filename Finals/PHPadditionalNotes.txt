We will show you a simple code on how to connect in MYSQL and display its table using PHP.

<?php

$host = "localhost";
$user = "root";
$pass="";
$db = "company";


$mysqli = new mysqli($host, $user, $pass, $db);

//show tables
$result = $mysqli->query("SHOW TABLES from company");

while($tableName = mysqli_fetch_row($result))
{
    $table = $tableName[0];
    echo '<h3>' ,$table, '</h3>';
    $result2 = $mysqli->query("SELECT * FROM company.baguiocompany");
    if(mysqli_num_rows($result2))
    {
        echo '<table cellpadding = "1" cellspacing = "1" class "db-table">';
        echo '<tr><th>id</th><th>company name</th><th> location</th></tr>';
        while($row2 = mysqli_fetch_row($result2))
        {
            echo '<tr>';
            foreach ($row2 as $key=>$value)
            {
                echo '<td>',$value, '</td>';
            }
            echo '</tr>';
        }
        echo '</table><br />';
    }
}
?>
